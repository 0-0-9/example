function drawMyCanvas(){
    var canvas = document.getElementById("myCanvas");
    if (canvas.getContext) {
        var context = canvas.getContext("2d");
        context.clearRect(0, 0, 300, 250);
        //绘制矩形
        context.fillStyle = "rgb(165, 143, 197)"
        context.fillRect(60, 50, 150, 125);
        context.strokeStyle = "pink";
        context.strokeRect(70, 60, 130, 105);
        context.clearRect(90, 80, 90, 65);
    };
}
drawMyCanvas();

function drawMyCanvas(){
    var canvas = document.getElementById("myCanvas");
    if (canvas.getContext) {
        var context = canvas.getContext("2d");
        context.clearRect(0, 0, 300, 250);
        //绘制粉色矩形
        context.fillStyle = "pink";
        context.fillRect(90, 75, 120, 100);
        //绘制半透明紫色矩形
        context.fillStyle = "rgba(165, 143, 197,0.5)"
        context.fillRect(60, 50, 120, 100);
    };
}
drawMyCanvas();

function drawMyCanvas(){
    var canvas = document.getElementById("myCanvas");
    if (canvas.getContext) {
        var context = canvas.getContext("2d");
        context.clearRect(0, 0, 300, 250);
        context.beginPath();
        //绘制封闭半圆
        context.arc(150, 125, 100, Math.PI, false);
        context.strokeStyle = "blue";
        context.closePath();
        context.stroke();
    };
}
drawMyCanvas();
