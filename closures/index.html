JavaScript中的闭包 
var firstClosure,secondClosure;
function foo(){
    var x = 10;
    firstClosure = function (){
        return "firstClosure: " +(++x);
    }
    secondClosure = function (){
        return "secondClosure: " +(--x);
    }
    x = 20;
    alert(firstClosure());
}
foo();
alert(firstClosure());
alert(secondClosure());

function foo(k,max){
    var result = new Array();
    for (var i = 0; i < max; i++){
        result[i] = function(){
            alert(i);
        };
    }
    return result[k]();
}

function foo(k,max){
    var result = new Array();
    for (var i = 0; i < max; i++){
        result[i] = function(num){
            return function(){
                alert(num);
            };
        }(i);
    }
    return result[k]();
}

name = "this is window";
