var num = 0;
var max = 3;
var intervalId = null;

function incrementNumber(){
    num++;
    //当执行次数达到max，取消后续调用
    if(num === max){
        clearInterval(intervalId);
        alert("num: "+ num + ",Done!");
    }
}
intervalId = setInterval(incrementNumber,500);

var num = 0;
var max = 3;

function incrementNumber(){
    num++;
    //当执行次数没有达到max，设置另一次超时调用
    if(num < max){
        setTimeout(incrementNumber,500);
    }else{
        alert("num: "+ num + ",Done!");
    }
}
setTimeout(incrementNumber,500);
